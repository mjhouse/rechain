60 /* Update trust maps
61 */
62 void BlockChain::update_trust(){
63 
64  this->usr_trust.clear();
65  this->pub_trust.clear();
66 
67  std::map<std::string,std::string> ref;
68 
69  if(!this->blockchain.empty()){
70  std::string gen_key = this->blockchain.at(0)->get_data(0)->get_public_key();
71  this->usr_trust.insert( std::make_pair(gen_key,1.0f) );
72 
73  for(auto b : this->blockchain){
74  for(auto d : b->get_data()){
75  switch( d->get_data_type() ){
76  case DataType::Publication:
77 
78  // init a trust entry for the publisher
79  this->usr_trust.insert( std::make_pair( d->get_public_key(), 0.0f ) );
80 
81  // update the signature -> user reference
82  ref.insert( std::make_pair( d->get_signature(), d->get_public_key() ) );
83 
84  break;
85  case DataType::Signature:
86  {
87  // Get publication reference, the signer and signee
88  // public keys
89  std::string pubref = d->get_data_ref();
90  std::string signer = d->get_public_key();
91  std::string signee = ref[pubref];
92 
93  float ct = this->usr_trust[signer]/2.0f;// index will return 0.0f by default
94  if(ct && !signee.empty()){
95  this->usr_trust[signer] = ct; // signer loses half of trust
96  this->usr_trust[signee] += ct; // signee gains half of signers trust
97  this->pub_trust[pubref] += ct; // pubref gains half of signers trust
98  }
99  }
100  break;
101  }
102  }
103  }
104  }
105 }

107 /* Mine and add a block to the chain
108 */
109 std::string BlockChain::mine(){
110  std::string ret_hash;
111  if(this->current){
112  // Update trust on the block if it's a signature
113  this->current->set_trust( this->get_publication_trust() );
114 
115  // Check if the chain has a genesis block
116  if(this->blockchain.size() > 0){
117  this->current->set_previous(this->blockchain.back()->hash());
118  }
119 
120  // Mine the block until it has a valid hash
121  while(this->current->hash() > HASH_MAX){
122 
123  // Update the hashing variables
124  this->current->change_hash();
125 
126  }
127 
128  // Return the hash
129  ret_hash = this->current->hash();
130 
131  // Add to the chain
132  this->blockchain.push_back( this->current );
133  this->current = nullptr;
134 
135  // Update trust maps
136  this->update_trust();
137  }
138  return ret_hash;
139 }
