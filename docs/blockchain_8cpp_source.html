61 // update_trust
62 // Description:
63 // Update the trust map
64 // ----------------------------------------------------------------------------
65 void Blockchain::update_trust(){
66 
67  if(m_blockchain.size() > 0){
68 
69  // clear the trust map and set the maximum available
70  // trust to the length of the blockchain.
71  m_trust.clear();
72 
73  // lost trust is the trust that is given
74  // to the records. they can't spend it, so it
75  // is subtracted from max trust after the trust
76  // calculation.
77  double lost_trust = 0;
78 
79  // get the distribution list from the genesis record and
80  // split the max_trust between them.
81 
82  auto genesis = std::dynamic_pointer_cast<GenesisRecord>(m_blockchain[0]);
83 
84  // should have a genesis block at all times
85  assert(genesis);
86 
87  // get the owners of the blockchain and split the
88  // starting trust between them.
89  auto distribution = genesis->get_distribution();
90  max_trust = static_cast<double>(m_blockchain.size());
91 
92  double partial = max_trust/distribution.size();
93  for(auto& identifier : distribution){
94  m_trust.emplace(identifier,partial);
95  }
96 
97  // a reference map to link record hashes to authors
98  std::map<std::string,std::string> reference;
99 
100  for(auto& record : m_blockchain){
101 
102  switch(record->get_type()){
103 
104  case RecordType::Publication:
105  {
106 
107  auto pub_record = std::dynamic_pointer_cast<PublicationRecord>(record);
108 
109  if(pub_record){
110 
111  std::string author = pub_record->get_public_key();
112  std::string hash = pub_record->hash();
113 
114  // get a reference to the trust of both
115  // the author and the record
116  double& author_trust = m_trust[author];
117  double& record_trust = m_trust[hash];
118 
119  // check the author has trust to give
120  if(author_trust > 0){
121 
122  double half_trust = (author_trust/2);
123 
124  // give the record half of the author's
125  // trust
126  author_trust = half_trust;
127  record_trust = half_trust;
128 
129  // save the lost half that went
130  // to the record.
131  lost_trust += half_trust;
132 
133  // link the hash to the author
134  reference.emplace(hash,author);
135  }
136 
137  }
138 
139  }
140  break;
141 
142  case RecordType::Signature:
143  {
144 
145  auto sig_record = std::dynamic_pointer_cast<SignatureRecord>(record);
146 
147  if(sig_record){
148 
149  // get the signer and the signer's trust
150  std::string signer = sig_record->get_public_key();
151  double& signer_trust = m_trust[signer];
152 
153  // check that they have trust to give
154  if(signer_trust > 0){
155 
156  std::string hash = sig_record->get_record_hash();
157  std::string signee = reference[hash];
158 
159  // remove one-half of the signer's trust
160  double quarter_trust = (signer_trust/4);
161  signer_trust = quarter_trust*2;
162 
163  // signee and the record each get one
164  // quarter of the signer's trust
165  m_trust[signee] += quarter_trust;
166  m_trust[hash] += quarter_trust;
167 
168  // save the lost quarter that went
169  // to the record.
170  lost_trust += quarter_trust;
171 
172  }
173 
174  }
175 
176  }
177  break;
178 
179  default:
180  {
181  // genesis
182  }
183  break;
184 
185  }
186 
187  }
188 
189  // max_trust is used for normalizing trust values
190  // when they are requested. removing lost trust
191  // means that the normalization is relative to free
192  // trust in the system only.
193  max_trust -= lost_trust;
194 
195  }
196 }

209 
210  if(m_blockchain.size() > 0){
211 
212  auto& last = m_blockchain.back();

214 
215  }
216 
217  t_record->mine();
218 
219  if(t_record->is_valid()){
220 
221  m_blockchain.push_back(t_record);
222  // remote->send( t_record );
223 
224  RCINFO("record was published: " + hash);
225  return true;
226  }
227 
228  RCERROR("record was not valid: " + hash);

318 // Name:
319 // Blockchain::is_valid
320 // Description:
321 // Check that currently loaded blockchain is valid
322 // ----------------------------------------------------------------------------
323 bool Blockchain::is_valid(){
324  RCDEBUG("checking if blockchain is valid");
325 
326  std::set<std::string> references;
327  std::set<std::string> hashes;
328  std::string previous;
329 
330  for(auto& record : m_blockchain){
331 
332  if(record->get_previous() != previous){
333  RCERROR("record doesn't reference previous");
334  return false;
335  }
336 
337  if(!record->is_valid()){
338  RCERROR("record isn't valid");
339  return false;
340  }
341 
342  // update previous for the next iteration
343  previous = record->hash();
344 
345  switch(record->get_type()){
346 
347  case RecordType::Publication:
348  {
349 
350  auto pub_record = std::dynamic_pointer_cast<PublicationRecord>(record);
351 
352  if(!pub_record){
353  RCERROR("record type is publication, record is not");
354  return false;
355  }
356 
357  // try to insert the reference and hash
358  auto r_pair = references.insert(pub_record->get_reference());
359  auto h_pair = hashes.insert(pub_record->hash());
360 
361  // second item in pair is false if insert failed,
362  // and both pairs should insert successfully
363  if(!r_pair.second && !h_pair.second){
364  RCERROR("duplicate records in blockchain");
365  return false;
366  }
367 
368  }
369  break;
370 
371  case RecordType::Signature:
372  {
373 
374  auto sig_record = std::dynamic_pointer_cast<SignatureRecord>(record);
375 
376  if(!sig_record){
377  RCERROR("record type is signature, record is not");
378  return false;
379  }
380 
381  // try to find the referenced record by reference
382  auto it = hashes.find(sig_record->get_record_hash());
383 
384  if(it == hashes.end()){
385  RCERROR("signature doesn't reference pre-existing publication");
386  return false;
387  }
388 
389  }
390  break;
391  }
392  }
393 
394  // all tests passed
395  return true;
396 }
