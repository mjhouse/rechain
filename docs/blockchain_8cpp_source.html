79 /* Calculate the trust for the current chain
80 */
81 std::map<std::string,float> BlockChain::trust(){
82  // Will hold computed trust for public keys
83  std::map<std::string,float> key_trust;
84  std::map<std::string,std::string> pub_trust;
85 
86  // Iterators
87  std::vector<std::shared_ptr<Block>>::iterator b_it;
88  std::vector<std::shared_ptr<Data>>::iterator d_it;
89 
90  // Get the owners public key (public key of first block)
91  // and insert it with a value of 1
92  std::string gen_key = this->blockchain.at(0)->get_data(0)->get_public_key();
93  key_trust.insert( std::pair<std::string,float>(gen_key,1.0f) );
94 
95  // Iterate over each block and get the data
96  for(b_it = this->blockchain.begin(); b_it != this->blockchain.end(); ++b_it){
97  std::shared_ptr<Block> block = *b_it;
98  std::vector<std::shared_ptr<Data>> data = block->get_data();
99 
100  // Iterate over each Data object and calculate trust
101  for(d_it = data.begin(); d_it != data.end(); ++d_it ){
102  std::shared_ptr<Data> d = *d_it;
103 
104  switch(d->get_data_type()){
105  case DataType::Publication:
106  {
107  // If this is a publication Data object, then
108  // the public key is added to 'key_trust' if
109  // it isn't already there
110 
111  // Get the public key and the signature
112  std::string key = d->get_public_key();
113  std::string ref = d->get_signature();
114 
115  // Try to insert the public key with an initial
116  // value of '0.0'
117  key_trust.insert( std::make_pair(key,0.0f) );
118 
119  // Update the publication record linking the signature
120  // to the public key
121  pub_trust.insert( std::make_pair(ref,key) );
122  }
123  break;
124  case DataType::Signature:
125  {
126  // If this is a signature Data object, then half
127  // of the signers trust goes to the owner of the
128  // referenced document
129 
130  // Get the public keys of the document publisher and
131  // the signer
132  std::string ref_key = pub_trust[ d->get_data_ref() ];
133  std::string sig_key = d->get_public_key();
134 
135  // Get the current trust of each
136  float sig_trust = key_trust[ sig_key ];
137  float ref_trust = key_trust[ ref_key ];
138 
139  // Move half of the signers trust to the
140  // publisher
141  sig_trust = sig_trust/2;
142  ref_trust += sig_trust;
143 
144  // Update the key_trust map
145  key_trust[ref_key] = ref_trust;
146  key_trust[sig_key] = sig_trust;
147  }
148  break;
149  }
150  }
151  }
152  return key_trust;
153 }
