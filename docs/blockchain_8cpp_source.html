55 // update_trust
56 // Description:
57 // Update the trust map
58 // ----------------------------------------------------------------------------
59 void Blockchain::update_trust(){
60 
61  trust_map.clear();
62 
63  // get the owner and give them max_trust
64  std::string owner = blockchain[0]->get_public_key();
65  std::string genesis = blockchain[0]->get_reference();
66  trust_map.insert( std::make_pair(owner,Blockchain::max_trust) );
67 
68  // no blockchain or only genesis
69  if(blockchain.size() <= 1){
70  return;
71  }
72 
73  // this map links author keys to
74  // document references.
75  std::map<std::string,std::string> lookup;
76 
77  for(auto& record : blockchain){
78 
79  double transferred = 0;
80 
81  switch(record->type()){
82 
83  case Record::Publication:
84  {
85 
86  std::string author = record.get_public_key();
87  std::string document = record.get_reference();
88 
89  // add each document hash and author's public key to the reference map
90  lookup.insert(std::pair<std::string,std::string>(document,author));
91 
92  // try to insert author and document into trust_map
93  auto a_it = trust_map.insert( std::pair<std::string,double>(author,0) );
94  auto d_it = trust_map.insert( std::pair<std::string,double>(document,0) );
95 
96  // if the insert failed, then the author may have trust, and if
97  // the author has trust, give half to the document
98  if(!a_it.second && document != genesis){
99 
100  // get references to the actual iterators
101  auto a_entry = a_it.first;
102  auto d_entry = d_it.first;
103 
104  // split up trust
105  double amount = a_entry->second/2;
106  a_entry->second -= amount;
107  d_entry->second += amount;
108 
109  transferred += amount;
110  }
111 
112  }
113  break;
114 
115  case Record::Signature:
116  {
117 
118  // get the public keys and publication reference
119  std::string pubref = record.get_reference();
120 
121  // try to find the signer in the trust map- if they
122  // aren't found, then they haven't published anything.
123  auto it = trust_map.find(record.get_public_key());
124 
125  // check if the signer is published
126  if( it != trust_map.end() ){
127 
128  std::string signer = it->first;
129  std::string signee = reference[pubref];
130 
131  double trust = it->second;
132 
133  // get amount to transfer
134  double doc_amount = trust*0.25;
135  double usr_amount = trust*0.25;
136 
137  // if the new document trust will be larger than
138  // trust max, set it to 0.
139  if((trust_map[pubref] + doc_amount) > Blockchain::max_trust){
140  doc_amount = 0;
141  }
142 
143  // transfer trust to the signee
144  trust_map[signee] += usr_amount;
145  trust_map[pubref] += doc_amount;
146  trust_map[signer] -= (doc_amount + usr_amount);
147 
148  // update the total of transferred trust
149  transferred += doc_amount;
150  }
151  }
152  break;
153 
154  default:
155  break;
156 
157  }
158 
159  }
160 
161 }

286 // Name:
287 // Blockchain::is_valid
288 // Description:
289 // Check that currently loaded blockchain is valid
290 // ----------------------------------------------------------------------------
291 bool is_valid(){
292  RCDEBUG("checking if blockchain is valid");
293 
294  std::set<std::string> references;
295  std::string previous;
296 
297  for(auto& record : blockchain){
298 
299  if(record->get_previous() != previous){
300  RCERROR("record doesn't reference previous");
301  return false;
302  }
303 
304  if(!record->is_valid()){
305  RCERROR("record isn't valid");
306  return false;
307  }
308 
309  // update previous for the next iteration
310  previous = record->hash()
311 
312 
313  switch(record->type()){
314  case Record::Publication:
315  {
316 
317  // try to insert the reference into the set
318  auto pair = references.insert(record->get_reference());
319 
320  // second item in pair is false if insert failed
321  if(!pair.second){
322  RCERROR("duplicate records in blockchain");
323  return false;
324  }
325 
326  }
327  break;
328 
329  case Record::Signature:
330  {
331 
332  // try to find the referenced record by reference
333  auto it = references.find(record.get_reference());
334 
335  if(it == references.end()){
336  RCERROR("signature doesn't reference pre-existing publication");
337  return false;
338  }
339 
340  }
341  break;
342  }
343  }
344 
345  // all tests passed
346  return true;
347 };
