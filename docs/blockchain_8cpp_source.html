44 /* Update trust maps
45 */
46 void BlockChain::update_trust(){
47 
48  if(!this->blockchain.empty() && this->blockchain[0].size() > 0){
49 
50  std::map<std::string,std::string> references;
51  std::vector<Record> records;
52 
53  pub_trust.clear();
54  usr_trust.clear();
55 
56  int count = 1;
57  for(auto b : blockchain){
58  for(auto r : b){
59  switch(r.type()){
60  case DataType::Publication:
61  references.insert( std::make_pair(r.reference(),r.public_key()) );
62  break;
63  case DataType::Signature:
64  records.push_back(r);
65  count++;
66  break;
67  }
68  }
69  }
70 
71  float gt = (float)(count*2);
72  std::string owner = blockchain[0][0].public_key();
73  usr_trust.insert( std::make_pair(owner,gt) );
74 
75  for(auto r : records){
76  std::string pubref = r.reference();
77  std::string signer = r.public_key();
78  std::string signee = references[pubref];
79 
80  float ct = usr_trust[signer]/2.0f;
81  if(ct && !signee.empty()){
82  usr_trust[signer] = ct; // signer loses half of trust
83  usr_trust[signee] += ct; // signee gains half of signers trust
84  pub_trust[pubref] += ct; // pubref gains half of signers trust
85  }
86  }
87  }
88 }

165 /* Check if BlockChain is valid
166 */
167 bool BlockChain::valid(){
168  std::string previous;
169  std::set<std::string> pubs;
170 
171  for(auto b : blockchain){
172 
173  if(b.previous() != previous)
174  return false;
175  if(b.hash() > HASH_MAX)
176  return false;
177 
178  previous = b.hash();
179 
180  for(auto r : b){
181  if(!r.valid()) return false;
182 
183  switch(r.type()){
184  case DataType::Publication:
185  if(!pubs.insert(r.reference()).second)
186  return false;
187  break;
188 
189  case DataType::Signature:
190  {
191  // Try to find the referenced block by hash
192  auto b_it = this->find(r.block());
193  if(b_it == this->end())
194  return false;
195 
196  // Try to find the referenced publication
197  Block& b = *b_it;
198  auto r_it = b.find(r.reference());
199  if(r_it == b.end())
200  return false;
201  }
202  break;
203  }
204  }
205  }
206  return true;
207 }
