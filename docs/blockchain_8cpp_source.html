49 /* Update trust maps
50 */
51 void BlockChain::update_trust(){
52 
53  usr_trust.clear();
54  pub_trust.clear();
55 
56  if(!this->blockchain.empty() && this->blockchain[0].size() > 0){
57 
58  // this map stores references that link author keys to
59  // document references.
60  std::map<std::string,std::string> reference;
61 
62  // the first record in the first block is the owner, they
63  // get an initial amount of trust equal to 'MAX_TRUST'.
64  std::string owner = blockchain[0][0].public_key();
65  usr_trust.insert( std::make_pair(owner,MAX_TRUST) );
66 
67  // iterate the through all blocks and all records
68  // in each block
69  for(auto& block : blockchain){
70  // get block author here for transfered-total distribution
71  std::string miner = block.public_key();
72  unsigned int transferred = 0;
73 
74  for(auto& record : block){
75 
76  switch(record.type()){
77  case DataType::Publication:
78  {
79  std::string author = record.public_key();
80  std::string document = record.reference();
81 
82  // add each document hash and author's public key to the reference map
83  reference.insert(std::pair<std::string,std::string>(document,author));
84 
85  usr_trust.insert( std::pair<std::string,unsigned int>(author,0) );
86  }
87  break;
88  case DataType::Signature:
89  {
90  // get the public keys and publication reference
91  std::string pubref = record.reference();
92 
93  // try to find the signer in the trust map- if they
94  // aren't found, then they haven't published anything.
95  auto it = usr_trust.find(record.public_key());
96 
97  // check if the signer is published
98  if( it != usr_trust.end() ){
99 
100  std::string signer = it->first;
101  std::string signee = reference[pubref];
102 
103  unsigned int trust = it->second;
104 
105  // get amount to transfer
106  unsigned int doc_amount = floor(trust*0.25);
107  unsigned int usr_amount = floor(trust*0.25);
108 
109  // if they have 1 trust, then doc_amount will be
110  // 0, so set doc_amount equal to total trust. we
111  // don't care if users get 0.
112  if(doc_amount == 0)
113  amount = trust;
114 
115  // transfer trust to the signee
116  usr_trust[signee] += usr_amount;
117  pub_trust[pubref] += doc_amount;
118  usr_trust[signer] -= (doc_amount + usr_amount);
119 
120  unsigned int transferred += doc_amount;
121  }
122  }
123  break;
124 
125  default:
126  // may add other types of records
127  break;
128  }
129  }
130 
131  // update the miner's trust
132  usr_trust[miner] += transferred;
133  }
134  }
135 }

139 std::string BlockChain::mine( std::string pubkey ){
140  // Check if the chain has a genesis block

142  this->current.previous(this->blockchain.back().hash());

144

147 
148  // Add to the chain
149  this->blockchain.push_back( this->current );

153  this->update_trust();
154

212 /* Check if BlockChain is valid
213 */
214 bool BlockChain::valid(){
215  std::string previous;
216  std::set<std::string> pubs;
217 
218  for(auto b : blockchain){
219 
220  if(b.previous() != previous)
221  return false;
222  if(b.hash() > HASH_MAX)
223  return false;
224 
225  previous = b.hash();
226 
227  for(auto r : b){
228  if(!r.valid()) return false;
229 
230  switch(r.type()){
231  case DataType::Publication:
232  if(!pubs.insert(r.reference()).second)
233  return false;
234  break;
235 
236  case DataType::Signature:
237  {
238  // Try to find the referenced block by hash
239  auto b_it = this->find(r.block());
240  if(b_it == this->end())
241  return false;
242 
243  // Try to find the referenced publication
244  Block& b = *b_it;
245  auto r_it = b.find(r.reference());
246  if(r_it == b.end())
247  return false;
248  }
249  break;
250  }
251  }
252  }
253  return true;
254 }
