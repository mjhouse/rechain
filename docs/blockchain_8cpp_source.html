55 /* Update trust maps
56 */
57 void BlockChain::update_trust(){
58 
59  this->usr_trust.clear();
60  this->pub_trust.clear();
61 
62  std::map<std::string,std::string> ref;
63 
64  if(!this->blockchain.empty() && this->blockchain[0].size() > 0){
65  std::string gen_key = this->blockchain[0][0].get_public_key();
66  this->usr_trust.insert( std::make_pair(gen_key,1.0f) );
67 
68  for(auto b : this->blockchain){
69  for(auto d : b){
70  switch( d.get_data_type() ){
71  case DataType::Publication:
72 
73  // init a trust entry for the publisher
74  this->usr_trust.insert( std::make_pair( d.get_public_key(), 0.0f ) );
75 
76  // update the signature -> user reference
77  ref.insert( std::make_pair( d.get_signature(), d.get_public_key() ) );
78 
79  break;
80  case DataType::Signature:
81  {
82  // Get publication reference, the signer and signee
83  // public keys
84  std::string pubref = d.get_data_ref();
85  std::string signer = d.get_public_key();
86  std::string signee = ref[pubref];
87 
88  float ct = this->usr_trust[signer]/2.0f;// index will return 0.0f by default
89  if(ct && !signee.empty()){
90  this->usr_trust[signer] = ct; // signer loses half of trust
91  this->usr_trust[signee] += ct; // signee gains half of signers trust
92  this->pub_trust[pubref] += ct; // pubref gains half of signers trust
93  }
94  }
95  break;
96  }
97  }
98  }
99  }
100 }

102 /* Mine and add a block to the chain
103 */
104 std::string BlockChain::mine(){
105  std::string hash;
106  // Update trust on the block if it's a signature
107  for(auto d : this->current){
108  if(d.get_data_type() == DataType::Signature){
109  float t = this->pub_trust[d.get_public_key()];
110  d.set_trust( t );
111  }
112  }
113 
114  // Check if the chain has a genesis block
115  if(this->blockchain.size() > 0){
116  this->current.previous(this->blockchain.back().hash());
117  }
118 
119  // Get the hash to return
120  hash = this->current.mine();
121 
122  // Add to the chain
123  this->blockchain.push_back( this->current );
124  this->current = Block();
125 
126  // Update trust maps
127  this->update_trust();
128 
129  return hash;
130 }
