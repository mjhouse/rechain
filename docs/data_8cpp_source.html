62 bool Data::verify(){
63  // At a minimum, a Data object needs a public key and
64  // signature to be valid
65  if(!this->public_key.empty() && !this->signature.empty()){
66 
67  // Get the values from the Address
68  std::string d_ref = std::get<0>(this->address);
69  std::string b_ref = std::get<1>(this->address);
70  DataType type = std::get<2>(this->address);
71 
72  // There has to be a data reference
73  if(d_ref.empty()) return false;
74 
75  // If there is no block reference, type must be Publication
76  if(b_ref.empty() && type != DataType::Publication) return false;
77 
78  // Load the public key and check the signature
79  std::shared_ptr<PublicKey> key(PublicKey::load_string(this->public_key));
80  return key->verify( this );
81  }
82 
83  return false;
84 }
