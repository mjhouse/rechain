56 bool Manager::configure( Level level ){
57 
58  // create a new Config instance
59  config.reset(new Config());
60  if(!config->initialize()){
61  return false;
62  }
63 
64  // create a new Remote instance
65  remote.reset(new Remote());
66  if(!remote->initialize(config)){
67  return false;
68  }
69 
70  // verify or create home dir structure
71  if(!make_home())
72  return false;
73 
74  // configure logger
75  Logger::get()
76  .with( Log("console",STDOUT,level) )
77  .with( Log("log",config->setting("log"),Level::error) );
78 
79  std::string private_key_path = config->setting("private_key");
80  std::string public_key_path = config->setting("public_key");
81  std::string blockchain_path = config->setting("blockchain");
82 
83  if(!blockchain.load(blockchain_path)){
84  blockchain.save(blockchain_path);
85  }
86 
87  // if there is no private key or key file, create a new one
88  if(fs::exists(private_key_path)){
89  try {
90  private_key.reset(PrivateKey::load_file(private_key_path));
91  }
92  catch(std::invalid_argument& e){
93  rl::get().error("Private key is corrupted: " + private_key_path);
94  return false;
95  }
96  }
97  else {
98  private_key.reset(PrivateKey::empty());
99  private_key->generate();
100  private_key->save(private_key_path);
101  }
102 
103  // if there is no public key or key file, create a new one
104  if(fs::exists(public_key_path)){
105  try {
106  public_key.reset(PublicKey::load_file(public_key_path));
107  }
108  catch(const std::invalid_argument& e){
109  rl::get().error("Public key is corrupted: " + private_key_path);
110  return false;
111  }
112  }
113  else {
114  public_key.reset(PublicKey::empty());
115  public_key->generate(this->private_key.get());
116 
117  public_key->save(public_key_path);
118  }
119 
120  return true;
121 }
