43 /* Get the hash of this block
44 */
45 std::string Block::hash(){
46  std::string hash_data;
47 
48  // Get the hashable data for each Record block
49  std::for_each( records.begin(), records.end(),
50  [&hash_data]( Record r ){
51  hash_data.append( r.string(true) );
52  });
53 
54  hash_data.append( this->prev );
55  hash_data.append( std::to_string(this->nonce) );
56  hash_data.append( std::to_string(this->timestamp) );
57  hash_data.append( std::to_string(this->counter) );
58 
59  // Create a hash
60  CryptoPP::SHA256 hasher;
61  std::string new_hash;
62 
63  // Hash, hex encode and save to new_hash
64  CryptoPP::StringSource ss(hash_data,true,
65  new CryptoPP::HashFilter(hasher,
66  new CryptoPP::HexEncoder(
67  new CryptoPP::StringSink(new_hash))));
68  // Return new_hash
69  return new_hash;
70 }

72 /* Change the hashing variables
73 */
74 std::string Block::mine(){
75 
76  // Mine the block until it has a valid hash
77  while(this->hash() > HASH_MAX){
78  // Update the counter, and reset to 0 if it
79  // gets to the int max
80  if(this->counter >= UINT_MAX-1) this->counter = 0;
81  else this->counter++;
82 
83  // Update the timestamp
84  auto e = std::chrono::system_clock::now().time_since_epoch();
85  auto seconds = std::chrono::duration_cast<std::chrono::seconds>(e).count();
86  this->timestamp = (long)seconds;
87 
88  // Update the random nonce
89  CryptoPP::AutoSeededRandomPool rng;
90  this->nonce = CryptoPP::Integer(rng,
91  CryptoPP::Integer(1),
92  CryptoPP::Integer(LONG_MAX)).ConvertToLong();
93  }
94 
95  return this->hash();
96 
97 }
