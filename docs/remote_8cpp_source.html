87 unsigned int Remote::send( Record& record ){
88 
89  // serialize the record
90  std::ostringstream data;
91  cereal::JSONOutputArchive archive(data);
92  archive( record );
93  std::string message = data.str();
94 
95  unsigned int result = 0;
96 
97  // for each peer, send a POST request with the
98  // serialized record in the body
99  for( auto& peer : get_peers() ){
100  std::string address = peer.first;
101  std::string port = peer.second;
102 
103  // connect to the endpoint
104  boost::asio::io_service io_service;
105 
106  // get a query and endpoint iterator, and connect
107  // a socket to them
108  tcp::resolver resolver(io_service);
109  tcp::resolver::query query(address,port);
110  tcp::resolver::iterator endpoint_iterator = resolver.resolve(query);
111 
112  tcp::socket socket(io_service);
113 
114  boost::system::error_code ec;
115  boost::asio::connect(socket, endpoint_iterator,ec);
116 
117  if(ec.value() != ECONNREFUSED){
118  // push all the headers into the stream
119  boost::asio::streambuf request;
120  std::ostream request_stream(&request);
121 
122  request_stream << "POST / HTTP/1.1 \r\n";
123  request_stream << "Host: localhost:8080\r\n";
124  request_stream << "User-Agent: Rechain/1.0\r\n";
125  request_stream << "Content-Type: text/plain; charset=utf-8 \r\n";
126  request_stream << "Accept: */*\r\n";
127  request_stream << "Content-Length: " << message.length() << "\r\n";
128  request_stream << "Connection: close\r\n\r\n"; //NOTE THE Double line feed
129  request_stream << message;
130 
131  boost::asio::write(socket,request);
132 
133  // feed the response into a stream
134  boost::asio::streambuf response;
135  boost::asio::read_until(socket, response, "\r\n");
136  std::istream response_stream(&response);
137 
138  // get the response status code
139  std::string http_version;
140  unsigned int status_code;
141 
142  response_stream >> http_version;
143  response_stream >> status_code;
144 
145  // if any of the peers return 200, the send
146  // operation was a success
147  if(status_code == 200){
148  result++;
149  }
150  }
151 
152  }
153 
154  return result;
155 }
