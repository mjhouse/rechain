118 void Remote::send( Record& record ){
119 
120  // serialize the record
121  std::ostringstream data;
122 
123  {
124  cereal::JSONOutputArchive archive(data);
125  archive( record );
126  }
127 
128  std::string message = data.str();
129 
130  // for each peer, send a POST request with the
131  // serialized record in the body
132  for( auto& peer : config->get_peers() ){
133  std::string address = peer.first;
134  std::string port = peer.second;
135 
136  // connect to the endpoint
137  boost::asio::io_service io_service;
138 
139  // get a query and endpoint iterator, and connect
140  // a socket to them
141  tcp::resolver resolver(io_service);
142  tcp::resolver::query query(address,port);
143  tcp::resolver::iterator endpoint_iterator = resolver.resolve(query);
144 
145  tcp::socket socket(io_service);
146 
147  boost::system::error_code ec;
148  boost::asio::connect(socket, endpoint_iterator,ec);
149 
150  if(ec.value() != ECONNREFUSED){
151 
152  Request res;
153 
154  res.set_method("POST");
155  res.set_url("/");
156  res.set_property("Host","localhost:8080");
157  res.set_property("User-Agent","Rechain/1.0");
158  res.set_property("Content-Type","text/plain; charset=utf-8");
159  res.set_property("Accept","*/*");
160  res.set_property("Content-Length",std::to_string(message.length()));
161  res.set_property("Message-Type","Publish");
162  res.set_property("Connection","close");
163  res.set_body(message);
164 
165  res.write(socket);
166  }
167 
168  }
169 }
