88 std::string Record::mine(){
89 
90  // mine the record until it has a valid hash
91  while(hash() > HASH_MAX){
92 
93  // update the random nonce
94  CryptoPP::AutoSeededRandomPool rng;
95  m_nonce = CryptoPP::Integer(rng,
96  CryptoPP::Integer(1),
97  CryptoPP::Integer(LONG_MAX)).ConvertToLong();
98 
99  // update the timestamp
100  auto e = std::chrono::system_clock::now().time_since_epoch();
101  auto seconds = std::chrono::duration_cast<std::chrono::seconds>(e).count();
102  m_timestamp = (long)seconds;
103 
104  // update the counter
105  m_counter++;
106 
107  }
108 
109  return hash();
110 
111 }

127 bool Record::is_valid(){
128 
129  bool valid = true;
130 
131  try {
132 
133  // if the public key is bad this will throw
134  std::shared_ptr<PublicKey> key(PublicKey::load_string(m_public_key));
135 
136  // check if the signature is valid
137  if(!key->verify( *this )){
138  valid = false;
139  }
140 
141  // check if the Record has been mined
142  if(hash() > HASH_MAX){
143  valid = false;
144  }
145 
146  // check if the record refers to something
147  if(m_reference.empty()){
148  valid = false;
149  }
150 
151  } catch (const CryptoPP::InvalidArgument& e){
152  RCERROR(e.what());
153  valid = false;
154  }
155 
156  return valid;
157 }
