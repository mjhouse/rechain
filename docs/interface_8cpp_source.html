104  ("m,mine","Mine a block")

106  ("l,list","List published documents");

108  try {
109 
110  auto result = options.parse(this->argc,this->argv);
111 
112  if(!home.empty()){
113  std::string path = home + "/rechain.blockchain";
114  std::string priv = home + "/keys/rsa.private";
115  std::string publ = home + "/keys/rsa.public";
116 
117  BlockChain& blockchain = BlockChain::get_blockchain();
118 
119  if(!blockchain.load(path)){
120  rl::get().info("couldn't load blockchain from: " + path);
121  blockchain.save(path);
122  }
123 
124  private_key = PrivateKey::load_file(priv);
125  public_key = PublicKey::load_file(publ);
126  }
127  else {
128  rl::get().error("RECHAIN_HOME isn't set!");
129  exit(1);
130  }
131 
132  if(result.count("help")){
133  std::cout << options.help() << std::endl;
134  }
135  else {
136  if(result.count("publish"))
137  this->publish(result["p"].as<std::string>());
138 
139  if(result.count("mine"))
140  this->mine();
141 
142  if(result.count("sign"))
143  this->sign(result["s"].as<std::string>());
144 
145  if(result.count("list"))
146  this->list();
147  }

149  rl::get().error("Failed to parse command line arguments!");
