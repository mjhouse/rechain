68 bool Interface::sign( std::string s ){
69  rl::get().info("Adding signature for '" + s.substr(0,20) + "...' to blockchain");
70 
71  Record record;
72  for(auto b : blockchain){
73  for(auto r : b){
74  if(r.reference() == s){
75  record.reference(r.reference());
76  record.block(b.hash());
77  private_key->sign(record);
78  }
79  }
80  }
81 
82  if(record.valid()){
83  blockchain.add(record);
84  blockchain.save();
85  } else {
86  rl::get().error("Referenced publication doesn't exist");
87  return false;

113 void Interface::list(){
114  #define print(x)(std::cout << x << std::endl)
115  print(" ---- Blockchain ---- ");
116 
117  for(unsigned int i = 0; i < blockchain.size(); ++i){
118  print("Block #" + std::to_string(i) + ":");
119  for(unsigned int j = 0; j < blockchain[i].size(); ++j){
120  auto record = blockchain[i][j];
121  std::string msg;
122  std::string ref = record.reference();
123  std::string own = record.public_key();
124  std::string type = (record.type() == DataType::Publication) ? "Publication" : "Signature";
125  float trust = (type == "Publication") ? blockchain.trust(record.reference()) : 0.0f;
126 
127  if(j == 0 && i == 0)
128  msg = "\tRecord: " + ref + " (Genesis)";
129  else
130  msg = "\tRecord: " + ref;
131 
132  print(msg);
133  print("\tOwner: " + own.substr(50,ref.length()-3) + "...");
134  print("\t\tType: " + type);
135  print("\t\tTrust: " + std::to_string(trust));
136  }
137  }

140 int Interface::execute(){
141 
142  cxxopts::Options options("ReChain","The distributed research journal");
143  options.add_options()
144  ("h,help","Display this usage message")
145  ("v,version","Display version information")
146  ("p,publish","Publish a document",cxxopts::value<std::string>(),"<path>")
147  ("c,check","Validate the blockchain")
148  ("m,mine","Mine a block")
149  ("s,sign","Sign a published document",cxxopts::value<std::string>(),"<path>")
150  ("private_key","Make a private key active",cxxopts::value<std::string>(),"<path>")
151  ("public_key","Make a public key active",cxxopts::value<std::string>(),"<path>")
152  ("l,list","List published documents")
153  ("verbose","All logging output")
154  ("silent","No logging output");
155 
156 
157  try {
158 
159  auto result = options.parse(this->argc,this->argv);
160 
161  Level level;
162  if(result.count("verbose")) level = Level::error;
163  else if(result.count("silent")) level = Level::none;
164  else level = Level::info;
165 
166  Logger::get()
167  .with( Log("console",STDOUT,level) )
168  .with( Log("log",home + "/rechain.log",Level::error) );
169 
170  // Check RECHAIN_HOME
171  if(!home.empty()){
172  std::string path = home + "/rechain.blockchain";
173  if(!blockchain.load(path)){
174  blockchain.save(path);
175  }
176  }
177  else {
178  // or error out
179  rl::get().error("RECHAIN_HOME isn't set!");
180  return ERROR;
181  }
182 
183  // Check for help first
184  if(result.count("help")){
185  // Display help for command line options
186  std::cout << options.help() << std::endl;
187  return NOERR;
188  }
189  else {
190 
191  // Do other options
192  std::string publ = home + "/current.public";
193  std::string priv = home + "/current.private";
194 
195  // Try to load a private key from a given path
196  // or from the default path
197  try {
198  if(result.count("private_key")){
199  private_key.reset(PrivateKey::load_file(result["private_key"].as<std::string>()));
200  private_key->save(priv);
201  } else {
202  private_key.reset(PrivateKey::load_file(priv));
203  }
204  } catch (const CryptoPP::InvalidArgument& e){
205  rl::get().error(e.what());
206  return ERROR;
207  }
208 
209  // Try to load a public key from a given path
210  // or from the default path
211  try {
212  if(result.count("public_key")){
213  public_key.reset(PublicKey::load_file(result["public_key"].as<std::string>()));
214  public_key->save(publ);
215  } else {
216  public_key.reset(PublicKey::load_file(publ));
217  }
218  } catch (const CryptoPP::InvalidArgument& e){
219  rl::get().error(e.what());
220  return ERROR;
221  }
222 
223  // Publish a document to the BlockChain
224  if(result.count("publish")){
225  if(this->publish(result["p"].as<std::string>()))
226  return NOERR;
227  else
228  return ERROR;
229  }
230 
231  // Check the BlockChain integrity
232  if(result.count("check")){
233  if(this->check())
234  return NOERR;
235  else
236  return ERROR;
237  }
238 
239  // Mine the current Block
240  if(result.count("mine")){
241  if(this->mine())
242  return NOERR;
243  else
244  return ERROR;
245  }
246 
247  // Sign a previously published Record
248  if(result.count("sign")){
249  if(this->sign(result["s"].as<std::string>()))
250  return NOERR;
251  else
252  return ERROR;
253 
254  }
255 
256  // Display the current BlockChain state
257  if(result.count("list")){
258  this->list();
259  return NOERR;
260  }
261 
262  // Display the current version
263  if(result.count("version")){
264  std::cout << "ReChain v" << RECHAIN_VERSION << " (GNU GPL v3) "
265  << "The distributed research journal" << std::endl;
266  return NOERR;
267  }
268  }
269  } catch (const cxxopts::OptionException& e){
270  rl::get().error("Failed to parse command line arguments!");
271  return ERROR;
272  }
273 
274  return NOERR;
275 }
